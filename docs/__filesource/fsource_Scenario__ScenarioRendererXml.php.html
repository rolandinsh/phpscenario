  50.     /**
  51.      * Sets the xslt parameter
  52.      *
  53.      * Alias for setParam('xslt',$translator). If the translator is an instance
  54.      * of DOMDocument it will be used in the XML translation by itself.
  55.      * If $translator starts with a slash, it will be interpreted as a path.
  56.      * Otherwise, the file is assumed to reside within the Scenario/Reporter/Xslt/
  57.      * folder, as {$translator}.xsl
  58.      *
  59.      * @param mixed $name Can be set to either a string or a DOMDocument (XSL)
  60.      */
  61.     public function setTranslator($translator) {
  62.  
  63.         if (is_string($translator) || $translator instanceof DOMDocument) {
  64.             $this->setParam('xslt', $translator);
  65.             return;
  66.         }
  67.  
  68.         require_once 'Scenario/Exception.php';
  69.         throw new Scenario_Exception('Translator must be a string or instance of DOMDocument.');
  70.     }

  81.     /**
  82.      * Render a Scenario_ResultSet
  83.      *
  84.      * Renders a result set in XML. To render multiple experiments, merge the
  85.      * collections, or render them individually.
  86.      *
  87.      * @param Scenario_ResultSet $results   The resultset to render.
  88.      * @param bool $capture     (optional) Whether to capture the results rather than render them to output.
  89.      * @return string   The rendered result document as a string.
  90.      */
  91.     public function renderSet(Scenario_ResultSet $results, $capture = false) {
  92.         $results = $this->resultsAsXml($results);
  93.  
  94.         if ($this->getTranslator() == self::AS_XML) {
  95.             $output = $results->saveXML();
  96.             if (!$capture)
  97.                 echo $output;
  98.             return $output;
  99.         }
  100.  
  101.         $xsl = $this->getTranslator();
  102.         if (is_string($xsl))
  103.             $xsl = $this->getXslt($xsl);
  104.  
  105.         if (!($xsl instanceof DOMDocument)) {
  106.             require_once 'Scenario/Exception.php';
  107.             throw new Scenario_Exception('Translator must be set to a string or an instance of DOMDocument.');
  108.         }
  109.  
  110.         $output = $this->translate($results, $xsl);
  111.         
  112.         if (!$capture)
  113.             echo $output;
  114.  
  115.         return $output;
  116.     }

  118.     /**
  119.      * Translates the provided xml and xsl DOMDocuments to string
  120.      *
  121.      * @param DOMDocument $xml XML to use in the translation.
  122.      * @param DOMDocument $xslt XSL Stylesheet to apply to the XML.
  123.      * @return string Output of the translation.
  124.      */
  125.     public function translate(DOMDocument $xml, DOMDocument $xslt) {
  126.         $renderer = new XSLTProcessor;
  127.         $renderer->importStylesheet($xslt);
  128.         $output = $renderer->transformToXml($xml);
  129.  
  130.         return $output;
  131.     }

  133.     /**
  134.      * Translates a result or resultset into an XML DOMDocument.
  135.      *
  136.      * @param Scenario_Result|Scenario_ResultSet$resultSet 
  137.      * @return DOMDocument 
  138.      */
  139.     public function resultsAsXml($resultSet) {
  140.         $xml = new XMLWriter();
  141.         $xml->openMemory();
  142.  
  143.         $data = $this->getDataFromResults($resultSet);
  144.  
  145.         $xml->startElement('experimentresults'); // root
  146.         $xml->writeAttribute('totalresults', $data['totalRecords']);
  147.  
  148.         foreach($data['results'] as $expid => $results) {
  149.  
  150.             $xml->startElement('experiment');
  151.  
  152.             $controlName = 'default'; // maybe make this an option eventually?
  153.             $cTotal = $results[$controlName]['total'];
  154.             $cComplete = $results[$controlName]['completed'];
  155.             $cRate = floatval($cComplete) / floatval($cTotal);
  156.  
  157.             $xml->writeAttribute('name', $expid);
  158.  
  159.             $eTotal = $results['_total'];
  160.             $xml->writeAttribute('total', $eTotal);
  161.  
  162.             foreach ($results as $key => $val) {
  163.  
  164.                 if ($key != '_total') {
  165.                     $tTotal = $val['total'];
  166.                     $tComplete = $val['completed'];
  167.                     $tRate = floatval($tComplete) / floatval($tTotal);
  168.                     
  169.                     $xml->startElement('treatment');
  170.                     $xml->writeAttribute('name', $key);
  171.  
  172.                     if ($key == $controlName) {
  173.                         $xml->writeAttribute('control', 'control');
  174.                     }
  175.  
  176.                     $xml->startElement('rawdata');
  177.                     $xml->writeAttribute('total', $tTotal);
  178.                     $xml->writeAttribute('completed', $tComplete);
  179.                     $xml->endElement(); // rawdata
  180.  
  181.                     $xml->startElement('statistics');
  182.  
  183.                     if ($key != $controlName) {
  184.                         $zscore = ($tRate - $cRate) / sqrt( ( ($tRate * (1.0 - $tRate)) / floatval($tTotal)) + ( ($cRate * (1.0 - $cRate)) / floatval($cTotal)) );
  185.                         $xml->writeAttribute('zscore', $zscore);
  186.                     }
  187.  
  188.                     $xml->writeAttribute('conversion', $tRate * 100.0);
  189.  
  190.                     $xml->endElement(); // statistics
  191.  
  192.                     $xml->endElement(); // treatment
  193.                 }
  194.                 
  195.             }
  196.             
  197.             $xml->endElement(); // experiment
  198.         }
  199.  
  200.         $xml->endElement(); // experimentresults (root)
  201.  
  202.         $dom = new DOMDocument;
  203.         $dom->loadXML($xml->outputMemory());
  204.         return $dom;
  205.     }

  207.     /**
  208.      * Get an array that summarizes the result set.
  209.      *
  210.      * Compiles resultset data into an array of summarized data, including
  211.      * number of results and number completed for each experiment and treatment.
  212.      *
  213.      * @param Scenario_ResultSet $resultSet 
  214.      * @return array 
  215.      */
  216.     private function getDataFromResults($resultSet) {
  217.  
  218.         if ($resultSet instanceof Scenario_Result) {
  219.             require_once 'Scenario/ResultSet.php';
  220.             $resultSet = new Scenario_ResultSet(array($resultSet));
  221.         }
  222.  
  223.         if (!($resultSet instanceof Scenario_ResultSet)) {
  224.             require_once 'Scenario/Exception.php';
  225.             throw new Scenario_Exception('resultsAsXml only accepts Scenario_Result and Scenario_ResultSet objects.');
  226.         }
  227.  
  228.         $data = array(
  229.             'results' => array(),
  230.             'totalRecords' => 0
  231.         );
  232.  
  233.         foreach($resultSet as $result) {
  234.             if ($result instanceof Scenario_Result) {
  235.                 $expid = $result->getExperiment()->getExperimentID();
  236.                 $treatment = $result->getTreatment()->getName();
  237.                 if (!array_key_exists($expid, $data['results'])) {
  238.                     $data['results'][$expid] = array(
  239.                         '_total' => 0
  240.                     );
  241.                 }
  242.                 if (!array_key_exists($treatment, $data['results'][$expid])) {
  243.                     $data['results'][$expid][$treatment] = array(
  244.                         'total' => 0,
  245.                         'completed' => 0,
  246.                     );
  247.                 }
  248.                 $data['results'][$expid][$treatment]['total']++;
  249.                 if ($result->isCompleted())
  250.                     $data['results'][$expid][$treatment]['completed']++;
  251.                 $data['totalRecords']++;
  252.                 $data['results'][$expid]['_total']++;
  253.             } else {
  254.                 require_once 'Scenario/Exception.php';
  255.                 throw new Scenario_Exception('Scenario_ResultSet must contain only Scenario_Result objects.');
  256.             }
  257.         }
  258.  
  259.         return $data;
  260.     }

  262.     /**
  263.      * Load an XSL file from Xslt subdir.
  264.      *
  265.      * @param string $name Name of the XSL file to load. Should be a filename within Scenario/Renderer/Xslt/
  266.      * @return DOMDocument 
  267.      */
  268.     public function getXslt($name) {
  269.         $xslPath = realpath(dirname(__FILE__) . DIRECTORY_SEPARATOR . 'Xslt' . DIRECTORY_SEPARATOR . ucfirst($name) . '.xsl');
  270.         if (!file_exists($xslPath)) {
  271.             require_once 'Scenario/Exception.php';
  272.             throw new Scenario_Exception('XSL file "'.$xslPath.'" does not exist.');
  273.         }
  274.         $xsl = new DOMDocument;
  275.         $xsl->load($xslPath);
  276.         return $xsl;
  277.     }
