  93.     public static function Treatment($experiment, Scenario_Identity $id = null, $create = true) {
  94.         // null check
  95.         if ($experiment === null) {
  96.             require_once 'Scenario/Exception.php';
  97.             throw new Scenario_Exception('Experiment parameter must be a string or instance of Scenario_Experiment.');
  98.         }
  99.  
  100.         // get the actual experiment for the string value
  101.         if (is_string($experiment))
  102.             $experiment = self::Experiment($experiment);
  103.  
  104.         // They didn't provide an identity, so grab one from the default (session).
  105.         if ($id == null) {
  106.             $id = self::Identity();
  107.         } else if (!($id instanceof Scenario_Identity)) {
  108.             require_once 'Scenario/Exception.php';
  109.             throw new Scenario_Exception('Identity provided must be an instance of Scenario_Identity.');
  110.         }
  111.  
  112.         if (!($experiment instanceof Scenario_Experiment)) {
  113.             require_once 'Scenario/Exception.php';
  114.             throw new Scenario_Exception('Experiment must be an instance of Scenario_Experiment.');
  115.         }
  116.  
  117.         return $experiment->getTreatment($id, $create);
  118.     }

  363.     /**
  364.      * Retrieve an experiment by name.
  365.      * 
  366.      * Gets an experiement by the specified name, first checking for the existence
  367.      * of a class in the format [classPrefix].ucfirst(name), and defaulting to
  368.      * Scenario_Experiment if unable to find a custom class. Custom classes may
  369.      * be used to alter the behavior of experiments, particularly in regards to
  370.      * treatment selection.
  371.      *
  372.      * @param string $experimentname 
  373.      * @return Scenario_Experiment 
  374.      */
  375.     public function getExperiment($experimentname) {
  376.         $prefix = $this->getOption('classPrefix');
  377.  
  378.         $className = $experimentname;
  379.         
  380.         if ($prefix != null) $className = $prefix . ucfirst($className);
  381.  
  382.         if (class_exists($className) && is_subclass_of($className, 'Scenario_Experiment')) {
  383.  
  384.             // instantiate the custom class
  385.             return new $className($experimentname);
  386.  
  387.         } else if ($this->getAdapter() !== null) {
  388.  
  389.             // failing custom creation, we'd much prefer to get it from the database if we can...
  390.             $exp = $this->getAdapter()->GetExperimentByName($experimentname);
  391.             if ($exp != null) return $exp;
  392.  
  393.         }
  394.         
  395.         // default to an experiment with no row ID.
  396.         require_once 'Scenario/Experiment.php';
  397.         return new Scenario_Experiment($experimentname);
  398.     }

  421.     public function getOption($str) {
  422.         $optStr = explode('.', $str);
  423.         if (array_key_exists($optStr[0], $this->_config)) {
  424.             $out = $this->_config[$optStr[0]];
  425.             for ($i = 1, $c = count($optStr); $i < $c; $i++) {
  426.                 if (!array_key_exists($optStr[$i], $out))
  427.                     return null;
  428.                 $out = $out[$optStr[$i]];
  429.             }
  430.             return $out;
  431.         }
  432.         return null;
  433.     }
