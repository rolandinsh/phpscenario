  121.         if ($this->_dbAdapter == null) {
  122.             $settings = $this->getCore()->getOption('db');
  123.  
  124.             if (!array_key_exists('uri', $settings)) {
  125.                 require_once 'Scenario/Data/Exception.php';
  126.                 throw new Scenario_Data_Exception('db.uri key not found in core settings.');
  127.             }
  128.

  133.             $this->_dbAdapter = Zend_Db::factory($dbParse['scheme'], array(
  134.                 'host'      => $dbParse['host'],
  135.                 'dbname'    => trim($dbParse['path'], "/\\ "),
  136.                 'username'  => $dbParse['user'],
  137.                 'password'  => $dbParse['pass']

  188.             require_once 'Scenario/Data/Exception.php';

  190.         }
  191.  
  192.         try {

  194.  
  195.             $this->getDbAdapter()->delete(
  196.                 $this->usersTreatmentsTable,
  197.                 array(
  198.                     $this->quote('experiment_id = ?', $experiment->getRowID()),
  199.                     $this->quote('treatment_id = ?', $treatment->getRowId())
  200.                 )
  201.             );
  202.             
  203.             if ($delete) {
  204.                 $this->getDbAdapter()->delete(
  205.                     $this->treatmentsTable,

  209.

  211.         } catch (Exception $e) {

  397.         if ($experiment->getRowID() == null)
  398.             $experiment = $this->GetExperimentByName($experiment->getExperimentID());
  399.

  401.             require_once 'Scenario/Data/Exception.php';

  403.         }
  404.  
  405.         $treatment = $this->GetTreatmentForIdentity($experiment, $id);
  406.         
  407.         if ($treatment !== null) {
  408.             // UPDATE EXISTING
  409.             $treatmentID = $this->GetTreatmentID($experiment, $value->getName());
  410.             if ($treatmentID == null) {
  411.                 $this->AddTreatment($experiment, $value->getName());
  412.                 $treatmentID = $this->GetTreatmentID($experiment, $value->getName());
  413.             }
  414.             if ($treatmentID !== null) {
  415.                 $result = $this->getDbAdapter()->update($this->usersTreatmentsTable,
  416.                     array('treatment_id'=>$treatmentID),
  417.                     $this->quote('identity = ? AND experiment_id = ?', array($id->getDbIdent(), $experiment->getRowID()))
  418.                 );
  419.             }
  420.         } else {
  421.             // CREATE NEW
  422.             
  423.             // check to see if the treatment specified actually exists & create it if not
  424.             $check = $this->GetTreatmentByName($experiment, $value->getName());
  425.             if ($check === null) { // treatment specified does not exist, must be created
  426.                 if ($this->AddTreatment($experiment, $value->getName())) {
