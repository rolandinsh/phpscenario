  160.     public function ClearTreatment($experiment, $treatment, $delete = false) {
  161.         if (is_string($experiment))
  162.             $experiment = $this->GetExperimentByName($experiment);
  163.         if (!($experiment instanceof Scenario_Experiment)) {
  164.             require_once 'Scenario/Data/Exception.php';

  167.  
  168.         if (is_string($treatment))
  169.             $treatment = $this->GetTreatmentByName($experiment, $treatment);
  170.         if (!($treatment instanceof Scenario_Treatment)) {
  171.             require_once 'Scenario/Data/Exception.php';
  172.             throw new Scenario_Data_Exception('Treatment is not an instance of Scenario_Treatment.');
  173.         }
  174.  
  175.         try {
  176.             $this->getDbAdapter()->beginTransaction();
  177.  
  178.             $sql = 'DELETE FROM '.$this->usersTreatmentsTable.' WHERE experiment_id = ? AND treatment_id = ?';
  179.             $this->prepare($sql, array($experiment->getRowID(), $treatment->getRowId()));
  180.  
  181.             if ($delete) {
  182.                 $otherSql = 'DELETE FROM '.$this->treatmentsTable.' WHERE id = ?';
  183.                 $this->prepare($otherSql, $treatment->getRowId());
  184.             }
  185.  
  186.             $this->getDbAdapter()->commit();
  187.         } catch (Exception $e) {
  188.             require_once 'Scenario/Data/Exception.php';

  363.     /**
  364.      * Set a Experiment/id combo to a specific treatment
  365.      *

  370.     public function SetTreatment(Scenario_Experiment $experiment, Scenario_Treatment $value, Scenario_Identity $id) {
  371.         if ($experiment->getRowID() == null)
  372.             $experiment = $this->GetExperimentByName($experiment->getExperimentID());
  373.  
  374.         if ($experiment === null || !is_int($experiment->getRowID())) {
  375.  
  376.             /**
  377.              * @see Scenario_Data_Exception
  378.              */
  379.             require_once 'Scenario/Data/Exception.php';
  380.             throw new Scenario_Data_Exception('Experiment does not exist.');
  381.         }
  382.  
  383.         $treatment = $this->GetTreatmentForIdentity($experiment, $id);
  384.  
  385.         if ($treatment !== null) {
  386.             // UPDATE EXISTING
  387.             $treatmentID = $this->GetTreatmentID($experiment, $value->getName());
  388.             if ($treatmentID == null) {
  389.                 $this->AddTreatment($experiment, $value->getName());
  390.                 $treatmentID = $this->GetTreatmentID($experiment, $value->getName());
  391.             }
  392.             if ($treatmentID !== null) {
  393.                 $sql = 'UPDATE `'.$this->usersTreatmentsTable.'` SET '.
  394.                         'treatment_id = ? WHERE identity = ? AND experiment_id = ?';
  395.                 $results = $this->prepare($sql, array($treatmentID, $id->getDbIdent(), $experiment->getRowID()));
  396.             }
  397.         } else {
  398.             // CREATE NEW
  399.  
  400.             // check to see if the treatment specified actually exists & create it if not
  401.             $check = $this->GetTreatmentByName($experiment, $value->getName());
  402.             if ($check === null) { // treatment specified does not exist, must be created
  403.                 if ($this->AddTreatment($experiment, $value->getName())) {
  404.                     $check = $this->GetTreatmentByName($experiment, $value->getName()); // should exist now
  405.                 } else {
  406.                     /**
  407.                      * @see Scenario_Data_Exception
  408.                      */
  409.                     require_once 'Scenario/Data/Exception.php';
  410.                     throw new Scenario_Data_Exception('Scenario was unable to create a new treatment record.');
  411.                 }
  412.             }
  413.  
  414.             // insert the new treatment/identity record
  415.             if ($check !== null) { // sanity
  416.  
  417.                 $sql = 'INSERT INTO `'.$this->usersTreatmentsTable.'` (identity, treatment_id, experiment_id, completed) '
  418.                         .'VALUES (?, ?, ?, ?)';
  419.                 $result = $this->prepare($sql, array(
  420.                     $id->getDbIdent(),
  421.                     $check->getRowId(),
  422.                     $experiment->getRowID(),
  423.                     0
  424.                 ));
  425.             }
  426.
